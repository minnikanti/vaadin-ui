import com.vaadin.flow.router.Route;
import com.vaadin.flow.spring.annotation.RouteScope;
import com.vaadin.flow.spring.annotation.SpringComponent;
import org.springframework.core.annotation.AliasFor;

import java.lang.annotation.ElementType;
import java.lang.annotation.Retention;
import java.lang.annotation.RetentionPolicy;
import java.lang.annotation.Target;

@Target(ElementType.TYPE)
@Retention(RetentionPolicy.RUNTIME)
@Route
@SpringComponent
@RouteScope
public @interface FreshRoute {
    
    @AliasFor(annotation = Route.class, attribute = "value")
    String value() default "";
    
    @AliasFor(annotation = Route.class, attribute = "layout")
    Class<? extends com.vaadin.flow.component.Component> layout() 
        default com.vaadin.flow.component.Component.class;
    
    @AliasFor(annotation = Route.class, attribute = "registerAtStartup")
    boolean registerAtStartup() default true;
}
